\item Aufgabe
\begin{lstlisting}
public int gcd(int x, int y) {
	int tmp;
    while (y != 0) {
    	tmp = x % y;
        x = y;
        y = tmp;
    }
    return x;
}

public int lcm(int n, int m) {
	return (n * m) / gcd(n , m);
}
\end{lstlisting}

\end{enumerate}
